|^XmlSameAs|''Checks the two lots of xml are the same''|${sameAs}|
|^XmlSameAsWithNameSpace|''Checks the two lots of xml are the same''|''but where name spaces are used''|
|^XmlSimilarTo|''Checks the two lots of xml are the same''|${similarTo}|
|>TransformWith|''Transforms the given xml with the xslt''|${transform}|
|>XpathExistsIn|''Checks for the existence of the element referenced by the xpath expression''|${exists}|
|>XpathInIs|''Checks the value of the evaluated xpath expression''|${in}|
|>XpathInMatches|''Checks if value of the evaluated xpath expression matches the regular expression''|${match}|
|>VariousErrorMessage|''Show various error messages when things go wrong''|
